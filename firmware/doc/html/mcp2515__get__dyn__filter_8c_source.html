35 // get a filter
36 
37 uint8_t mcp2515_get_filter(uint8_t number, can_filter_t *filter)
38 {
39  uint8_t mask_address;
40  uint8_t filter_address;
41  uint8_t temp;
42  uint8_t mode = mcp2515_read_register(CANSTAT);
43 
44  if (number > 5)
45  return 0;
46 
47  // change to configuration mode
48  mcp2515_change_operation_mode( (1<<REQOP2) );
49 
50  if (number <= 1)
51  {
52  mask_address = RXM0SIDH;
53  temp = mcp2515_read_register(RXB0CTRL);
54  }
55  else
56  {
57  mask_address = RXM1SIDH;
58  temp = mcp2515_read_register(RXB1CTRL);
59  }
60 
61  temp &= (1<<RXM1)|(1<<RXM0);
62 
63  if (temp == 0) {
64  // filter and masks are disabled
65  #if SUPPORT_EXTENDED_CANID
66  filter->flags.extended = 0;
67  #endif
68  filter->flags.rtr = 0;
69  filter->mask = 0;
70  filter->id = 0;
71 
72  return 1;
73  }
74 
75  #if SUPPORT_EXTENDED_CANID
76  // transform bits so that they match the format from can.h
77  temp >>= 5;
78  temp = ~temp;
79  if (temp & 1) temp = 0x3;
80 
81  filter->flags.extended = temp;
82  #endif
83 
84  // read mask
85  RESET(MCP2515_CS);
86  spi_putc(SPI_READ);
87  spi_putc(mask_address);
88  mcp2515_read_id(&filter->mask);
89  SET(MCP2515_CS);
90 
91  if (number <= 2)
92  {
93  filter_address = RXF0SIDH + number * 4;
94  }
95  else
96  {
97  filter_address = RXF3SIDH + (number - 3) * 4;
98  }
99 
100  // read filter
101  RESET(MCP2515_CS);
102  spi_putc(SPI_READ);
103  spi_putc(filter_address);
104  mcp2515_read_id(&filter->id);
105  SET(MCP2515_CS);
106 
107  // restore previous mode
108  mcp2515_change_operation_mode( mode );
109 
110  return 1;
111 }

113 #endif // SUPPORT_FOR_MCP2515__
