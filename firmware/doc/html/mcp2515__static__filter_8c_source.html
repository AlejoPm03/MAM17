37 void mcp2515_static_filter(const uint8_t *filter)
38 {
39  // change to configuration mode
40  mcp2515_bit_modify(CANCTRL, 0xe0, (1<<REQOP2));
41  while ((mcp2515_read_register(CANSTAT) & 0xe0) != (1<<REQOP2))
42  ;
43 
44  mcp2515_write_register(RXB0CTRL, (1<<BUKT));
45  mcp2515_write_register(RXB1CTRL, 0);
46 
47  uint8_t i, j;
48  for (i = 0; i < 0x30; i += 0x10)
49  {
50  RESET(MCP2515_CS);
51  spi_putc(SPI_WRITE);
52  spi_putc(i);
53 
54  for (j = 0; j < 12; j++)
55  {
56  if (i == 0x20 && j >= 0x08)
57  break;
58 
59  spi_putc(pgm_read_byte(filter++));
60  }
61  SET(MCP2515_CS);
62  }
63 
64  mcp2515_bit_modify(CANCTRL, 0xe0, 0);
65 }
66 
67 #endif // SUPPORT_FOR_MCP2515__
